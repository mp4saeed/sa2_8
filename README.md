حالا کد های زیر رو کپی کنید و بزنید.
**********************************************************
cd $HOME && git clone https://github.com/mp4saeed/sa2_8.git && cd sa2_8 && chmod +x energy.sh && ./energy.sh install && ./energy.sh
************************************************************

***************************************************
cd energy
